    for (var i = 0; i < str.length; i++) {
        for (var j = str.length; j > i; j--){

                if (subStr.length > maxStr.length) {
                    maxStr = subStr;
                }
            }
        }
    }
    if (maxStr === str.substr(0,1)) return -1;
    return maxStr
}
